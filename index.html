X-View

A JavaScript library like React, but uses Web Components.

  • GitHub
  • NPM
  • Docs
var x = require("x-view");

class Button extends x.Component {
  render() {
    return (
      <button type="button">
        <content />
      </button>
    );
  }
}

x.register("x-button", Button);
x.render(document.body, (
  <x-button>
    <span>Hello!</span>
  </x-button>
));
<body>
  <x-button>
    <span>Hello!</span>
  </x-button>
</body>
